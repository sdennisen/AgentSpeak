113  public static Pair<Boolean, Set<IVariable<?>>> unifytrigger( final IUnifier p_unifier, final ITrigger p_source, final ITrigger p_target )
114  {
115  // filter for avoid duplicated instantiation on non-existing values / annotations
116  if ( !( ( p_source.getLiteral().emptyValues() == p_target.getLiteral().emptyValues() )
117  && ( p_source.getLiteral().emptyAnnotations() == p_target.getLiteral().emptyAnnotations() ) ) )
118  return new ImmutablePair<>( false, Collections.emptySet() );
119 
120  // unify variables, source trigger literal must be copied
121  final Set<IVariable<?>> l_variables = p_unifier.literal( p_target.getLiteral().deepcopy().<ILiteral>raw(), p_source.getLiteral() );
122 
123  // check for completely unification (of all variables)
124  return l_variables.size() == CCommon.variablefrequency( p_target.getLiteral() ).size()
125  ? new ImmutablePair<>( true, l_variables )
126  : new ImmutablePair<>( false, Collections.emptySet() );
127  }
