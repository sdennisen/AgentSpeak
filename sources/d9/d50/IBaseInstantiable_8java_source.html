118  @Override
119  public IFuzzyValue<Boolean> execute( final IContext p_context, final boolean p_parallel, final List<ITerm> p_argument, final List<ITerm> p_return,
120  final List<ITerm> p_annotation
121  )
122  {
123  // execution must be the first call, because all elements must be executed and iif the execution fails the @atomic flag can be checked,
124  // each item gets its own parameters, annotation and return stack, so it will be created locally, but the return list did not to be an "empty-list"
125  // because we need to allocate memory of any possible element, otherwise an unsupported operation exception is thrown
126  final List<IFuzzyValue<Boolean>> l_result = m_annotation.containsKey( IAnnotation.EType.PARALLEL )
127  ? this.executeparallel( p_context )
128  : this.executesequential( p_context );
129  // if atomic flag if exists use this for return value
130  return m_annotation.containsKey( IAnnotation.EType.ATOMIC )
131  ? CFuzzyValue.from( true )
132  : l_result.stream().collect( p_context.agent().fuzzy().getResultOperator() );
133  }

143  private List<IFuzzyValue<Boolean>> executesequential( final IContext p_context )
144  {
145  final List<IFuzzyValue<Boolean>> l_result = Collections.synchronizedList( new LinkedList<>() );
146 
147  m_action.stream()
148  .map( i -> {
149  final IFuzzyValue<Boolean> l_return = i.execute(
150  p_context, false, Collections.<ITerm>emptyList(), new LinkedList<>(), Collections.<ITerm>emptyList() );
151  l_result.add( l_return );
152  return p_context.agent().fuzzy().getDefuzzyfication().defuzzify( l_return );
153  } )
154  .filter( i -> !i )
155  .findFirst();
156 
157  return l_result;
158  }
