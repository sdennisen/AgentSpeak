220  CActionWrapper( final ILiteral p_literal, final Map<IPath, IAction> p_actions, final Multiset<IAction> p_scorecache )
221  {
222  // check parallel and inner execution
223  m_parallel = p_literal.hasAt();
224 
225 
226  // resolve action
227  m_action = p_actions.get( p_literal.fqnfunctor() );
228  if ( m_action == null )
229  throw new CIllegalArgumentException( org.lightjason.agentspeak.common.CCommon.languagestring( this, "actionunknown", p_literal ) );
230 
231  // check number of arguments and add action to the score cache
232  if ( p_literal.orderedvalues().count() < m_action.minimalArgumentNumber() )
233  throw new CIllegalArgumentException(
234  org.lightjason.agentspeak.common.CCommon.languagestring( this, "argumentnumber", p_literal, m_action.minimalArgumentNumber() ) );
235 
236  p_scorecache.add( m_action );
237 
238 
239  // resolve action arguments and annotation
240  m_arguments = Collections.unmodifiableMap(
241  this.createSubExecutions( p_literal.orderedvalues().collect( Collectors.toList() ), p_actions, p_scorecache ) );
242  m_annotation = Collections.unmodifiableMap(
243  this.createSubExecutions( p_literal.annotations().collect( Collectors.toSet() ), p_actions, p_scorecache ) );
244  }

296  @SuppressWarnings( "unchecked" )
297  private Map<Integer, IExecution> createSubExecutions( final Collection<? extends ITerm> p_elements, final Map<IPath, IAction> p_actions,
298  final Multiset<IAction> p_scorecache
299  )
300  {
301  // convert collection to list and build map with indices
302  final List<? extends ITerm> l_elements = new LinkedList<>( p_elements );
303  return IntStream.range( 0, l_elements.size() ).boxed().collect( Collectors.toMap( i -> i, i -> {
304  final ITerm l_term = l_elements.get( i );
305  return l_term instanceof ILiteral ? new CActionWrapper( (ILiteral) l_term, p_actions, p_scorecache ) : new CTermWrapper<>( l_term );
306  } ) );
307  }
