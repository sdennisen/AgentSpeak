48  public final <T extends ITerm> boolean unify( final Set<IVariable<?>> p_variables, final Stream<T> p_source, final Stream<T> p_target )
49  {
50  final List<T> l_target = p_target.collect( Collectors.toList() );
51  final List<T> l_source = p_source.collect( Collectors.toList() );
52 
53  if ( ( l_target.isEmpty() ) || ( l_source.isEmpty() ) )
54  return true;
55 
56  if ( l_target.size() != l_source.size() )
57  return false;
58 
59  return StreamUtils.zip(
60  l_source.stream(),
61  l_target.stream(),
62  ( s, t ) -> {
63 
64  // if s and t are variable create a realocated variable for backtracking
65  if ( ( t instanceof IVariable<?> ) && ( s instanceof IVariable<?> ) )
66  {
67  p_variables.add(
68  ( (IVariable<?>) t ).mutex()
69  ? new CRelocateMutexVariable<>( ( (IVariable<?>) t ).fqnfunctor(), (IVariable<?>) s )
70  : new CRelocateVariable<>( ( (IVariable<?>) t ).fqnfunctor(), (IVariable<?>) s )
71  );
72  return true;
73  }
74 
75  // if target type is a variable set the value
76  if ( t instanceof IVariable<?> )
77  {
78  p_variables.add( ( (IVariable<Object>) t ).set( s ) );
79  return true;
80  }
81 
82  // if both raw values -> equality check
83  if ( ( s instanceof IRawTerm<?> ) || ( t instanceof IRawTerm<?> ) )
84  return s.equals( t );
85 
86  // if a literal exists -> source and target literal must be equal with the functor -> recursive descent
87  if ( ( s instanceof ILiteral ) && ( t instanceof ILiteral ) )
88  {
89  final ILiteral l_sourceliteral = (ILiteral) s;
90  final ILiteral l_targetliteral = (ILiteral) t;
91 
92  if ( !l_sourceliteral.fqnfunctor().equals( l_targetliteral.fqnfunctor() ) )
93  return false;
94 
95  return this.unify( p_variables, l_sourceliteral.orderedvalues(), l_targetliteral.orderedvalues() );
96  }
97 
98  // otherwise false
99  return false;
100  }
101  ).allMatch( i -> i );
102  }
