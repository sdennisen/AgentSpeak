66  @SuppressWarnings( "unchecked" )
67  protected static IFuzzyValue<Boolean> execute( final IContext p_context, final ILiteral p_value, final boolean p_parallel )
68  {
69  // read current rules, if not exists execution fails
70  final Collection<IRule> l_rules = p_context.agent().rules().get( p_value.fqnfunctor() );
71  if ( l_rules == null )
72  return CFuzzyValue.from( false );
73 
74  // first step is the unification of the caller literal, so variables will be set from the current execution context
75  final ILiteral l_unified = p_value.allocate( p_context );
76 
77  // second step execute backtracking rules sequential / parallel
78  return (
79  p_parallel
80  ? l_rules.parallelStream()
81  : l_rules.stream()
82  ).map( i -> {
83 
84  // instantiate variables by unification of the rule literal
85  final Set<IVariable<?>> l_variables = p_context.agent().unifier().literal( i.getIdentifier(), l_unified );
86 
87  // execute rule
88  final IFuzzyValue<Boolean> l_return = i.execute(
89  i.instantiate( p_context.agent(), l_variables.stream() ),
90  false,
91  Collections.<ITerm>emptyList(),
92  Collections.<ITerm>emptyList(),
93  Collections.<ITerm>emptyList()
94  );
95 
96  // create rule result with fuzzy- and defuzzificated value and instantiate variable set
97  return new ImmutableTriple<>( p_context.agent().fuzzy().getDefuzzyfication().defuzzify( l_return ), l_return, l_variables );
98 
99  } )
100 
101  // find successfully ended rule
102  .filter( ImmutableTriple::getLeft )
103  .findFirst()
104 
105  // realocate rule instantiated variables back to execution context
106  .map( i -> {
107 
108  i.getRight().parallelStream()
109  .filter( j -> j instanceof IRelocateVariable )
110  .forEach( j -> ( (IRelocateVariable) j ).relocate() );
111 
112  return i.getMiddle();
113 
114  } )
115 
116  // otherwise rule fails (default behaviour)
117  .orElse( CFuzzyValue.from( false ) );
118  }
