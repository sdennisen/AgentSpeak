131  // create immutable structures
132  final Multimap<IPath, ILiteral> l_annotations = HashMultimap.create();
133  p_annotations.forEach( i -> l_annotations.put( i.fqnfunctor(), i ) );
134  m_annotations = ImmutableSetMultimap.copyOf( l_annotations );
135 
136  final Multimap<IPath, ITerm> l_values = LinkedListMultimap.create();
137  p_values.forEach( i -> l_values.put( i.fqnfunctor(), i ) );

139 
140  m_orderedvalues = Collections.unmodifiableList( new LinkedList<>( p_values ) );
141 
142  // calculates hash value
143  m_hash = m_functor.hashCode()
144  ^ IntStream.range( 0, m_orderedvalues.size() ).boxed().mapToInt( i -> ( i + 1 ) * m_orderedvalues.get( i ).hashCode() ).sum()
145  ^ m_annotations.values().stream().mapToInt( Object::hashCode ).sum()
146  ^ ( m_negated ? 0 : 55529 )
147  ^ ( m_at ? 0 : 8081 );

150  // calculates the structure hash value (Murmur3) of the value and annotation definition
151  // functor will be added iif no literal data exists ( hasher must be existing twice )
152  final String l_functor = p_functor.getPath();
153 
154  final Hasher l_valuehasher = CCommon.getTermHashing();
155  if ( m_orderedvalues.stream().filter( i -> i instanceof ILiteral ).map( i -> l_valuehasher.putInt( ( (ILiteral) i ).valuehash() ) ).count() == 0 )
156  {
157  l_valuehasher.putBoolean( m_negated );
158  l_valuehasher.putString( l_functor, Charsets.UTF_8 );
159  }
160 
161  final Hasher l_annotationhasher = CCommon.getTermHashing();
162  if ( m_annotations.values().stream().map( i -> l_annotationhasher.putInt( i.valuehash() ) ).count() == 0 )
163  {
164  l_annotationhasher.putBoolean( m_negated );
165  l_annotationhasher.putString( l_functor, Charsets.UTF_8 );
166  }
