137  @Override
138  public final IConsistency call() throws Exception
139  {
140  if ( m_data.size() < 2 )
141  return this;
142 
143  // get key list of map for addressing elements in the correct order
144  final ArrayList<IAgent<?>> l_keys = new ArrayList<>( m_data.keySet() );
145 
146  // calculate markov chain transition matrix
147  final DoubleMatrix2D l_matrix = new DenseDoubleMatrix2D( m_data.size(), m_data.size() );
148  IntStream.range( 0, l_keys.size() ).parallel().boxed().forEach( i -> {
149 
150  final IAgent<?> l_item = l_keys.get( i );
151  IntStream.range( i + 1, l_keys.size() ).boxed().forEach( j -> {
152 
153  final double l_value = this.getMetricValue( l_item, l_keys.get( j ) );
154  l_matrix.setQuick( i, j, l_value );
155  l_matrix.setQuick( j, i, l_value );
156 
157  } );
158 
159  // row-wise normalization for getting probabilities
160  final double l_norm = Algebra.DEFAULT.norm1( l_matrix.viewRow( i ) );
161  if ( l_norm != 0 )
162  l_matrix.viewRow( i ).assign( Mult.div( l_norm ) );
163 
164  // set epsilon slope for preventing periodic markov chains
165  l_matrix.setQuick( i, i, m_epsilon );
166  } );
167 
168  // check for a zero-matrix
169  final DoubleMatrix1D l_eigenvector = l_matrix.zSum() <= m_data.size() * m_epsilon
170  ? new DenseDoubleMatrix1D( m_data.size() )
171  : m_algorithm.getStationaryDistribution( m_iteration, l_matrix );
172 
173  // calculate the inverted probability and normalize with 1-norm
174  l_eigenvector.assign( PROBABILITYINVERT );
175  l_eigenvector.assign( Functions.div( Algebra.DEFAULT.norm1( l_eigenvector ) ) );
176 
177  // set consistency value for each entry and update statistic
178  m_statistic.clear();
179  IntStream.range( 0, l_keys.size() )
180  .boxed()
181  .forEach( i -> {
182  m_statistic.addValue( l_eigenvector.get( i ) );
183  m_data.put( l_keys.get( i ), l_eigenvector.get( i ) );
184  } );
185 
186  return this;
187  }

322  public final DoubleMatrix1D getStationaryDistribution( final int p_iteration, final DoubleMatrix2D p_matrix )
323  {

355  private static DoubleMatrix1D getLargestEigenvector( final DoubleMatrix2D p_matrix, final int p_iteration )
356  {
357  final DoubleMatrix1D l_probability = DoubleFactory1D.dense.random( p_matrix.rows() );
358  IntStream.range( 0, p_iteration ).forEach( i -> {
359  l_probability.assign( Algebra.DEFAULT.mult( p_matrix, l_probability ) );
360  l_probability.assign( Mult.div( Algebra.DEFAULT.norm2( l_probability ) ) );
361  } );
362  return l_probability;

371  private static DoubleMatrix1D getLargestEigenvector( final DoubleMatrix2D p_matrix )
372  {
