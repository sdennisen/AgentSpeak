146  public static Stream<IAction> actionsFromPackage( final String... p_package )
147  {
148  return ( ( p_package == null ) || ( p_package.length == 0 )
149  ? Stream.of( MessageFormat.format( "{0}.{1}", PACKAGEROOT, "action.buildin" ) )
150  : Arrays.stream( p_package ) )
151  .flatMap( j -> {
152  try
153  {
154  return ClassPath.from( Thread.currentThread().getContextClassLoader() )
155  .getTopLevelClassesRecursive( j )
156  .parallelStream()
157  .map( ClassPath.ClassInfo::load )
158  .filter( i -> !Modifier.isAbstract( i.getModifiers() ) )
159  .filter( i -> !Modifier.isInterface( i.getModifiers() ) )
160  .filter( i -> Modifier.isPublic( i.getModifiers() ) )
161  .filter( IAction.class::isAssignableFrom )
162  .map( i -> {
163  try
164  {
165  return (IAction) i.newInstance();
166  }
167  catch ( final IllegalAccessException | InstantiationException l_exception )
168  {
169  LOGGER.warning( CCommon.languagestring( CCommon.class, "actioninstantiate", i, l_exception ) );
170  return null;
171  }
172  } )
173 
174  // action can be instantiate
175  .filter( Objects::nonNull )
176 
177  // check usable action name
178  .filter( CCommon::actionusable );
179  }
180  catch ( final IOException l_exception )
181  {
182  throw new UncheckedIOException( l_exception );
183  }
184  } );
185  }

229  private static boolean actionusable( final IAction p_action )
230  {
231  if ( ( p_action.name() == null ) || ( p_action.name().isEmpty() ) || ( p_action.name().get( 0 ).trim().isEmpty() ) )
232  {
233  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionnameempty" ) );
234  return false;
235  }
236 
237  if ( !Character.isLetter( p_action.name().get( 0 ).charAt( 0 ) ) )
238  {
239  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionletter", p_action ) );
240  return false;
241  }
242 
243  if ( !Character.isLowerCase( p_action.name().get( 0 ).charAt( 0 ) ) )
244  {
245  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionlowercase", p_action ) );
246  return false;
247  }
248 
249  if ( p_action.minimalArgumentNumber() < 0 )
250  {
251  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionargumentsnumber", p_action ) );
252  return false;
253  }
254 
255  return true;
256  }

267  private static Stream<Method> methods( final Class<?> p_class, final Class<?> p_root )
268  {
269  final Pair<Boolean, IAgentAction.EAccess> l_classannotation = CCommon.isActionClass( p_class );
270  if ( !l_classannotation.getLeft() )
271  return p_class.getSuperclass() == null
272  ? Stream.of()
273  : methods( p_class.getSuperclass(), p_root );
274 
275  final Predicate<Method> l_filter = IAgentAction.EAccess.WHITELIST.equals( l_classannotation.getRight() )
276  ? i -> !CCommon.isActionFiltered( i, p_root )
277  : i -> CCommon.isActionFiltered( i, p_root );
278 
279  return Stream.concat(
280  Arrays.stream( p_class.getDeclaredMethods() )
281  .parallel()
282  .map( i -> {
283  i.setAccessible( true );
284  return i;
285  } )
286  .filter( i -> !Modifier.isAbstract( i.getModifiers() ) )
287  .filter( i -> !Modifier.isInterface( i.getModifiers() ) )
288  .filter( i -> !Modifier.isNative( i.getModifiers() ) )
289  .filter( i -> !Modifier.isStatic( i.getModifiers() ) )
290  .filter( l_filter ),
291  methods( p_class.getSuperclass(), p_root )
292  );
293  }

301  private static Pair<Boolean, IAgentAction.EAccess> isActionClass( final Class<?> p_class )
302  {
303  if ( !p_class.isAnnotationPresent( IAgentAction.class ) )
304  return new ImmutablePair<>( false, IAgentAction.EAccess.BLACKLIST );

310  .parallel()
311  .anyMatch( p_class::equals )
