98  private static String httpdata( final String p_url ) throws IOException
99  {
100  final HttpURLConnection l_connection = (HttpURLConnection) new URL( p_url ).openConnection();
101 
102  // follow HTTP redirects
103  l_connection.setInstanceFollowRedirects( true );
104 
105  // set a HTTP-User-Agent if not exists
106  l_connection.setRequestProperty( "User-Agent",
107  ( System.getProperty( "http.agent" ) == null ) || ( System.getProperty( "http.agent" ).isEmpty() )
108  ? CCommon.PACKAGEROOT + CCommon.configuration().getString( "version" )
109  : System.getProperty( "http.agent" )
110  );
111 
112  // read stream data
113  final InputStream l_stream = l_connection.getInputStream();
114  final String l_return = CharStreams.toString(
115  new InputStreamReader(
116  l_stream,

118  ? Charsets.UTF_8
119  : Charset.forName( l_connection.getContentEncoding() )
120  )
121  );
122  Closeables.closeQuietly( l_stream );
123 
124  return l_return;
