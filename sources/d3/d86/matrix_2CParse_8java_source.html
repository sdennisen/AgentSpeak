96  private static double[][] parse( final IContext p_context, final String p_string )
97  {
98  final String[] l_rows = p_string.split( ";" );
99  final List<List<Double>> l_matrix = new ArrayList<>();
100 
101  final double[][] l_return = new double[l_rows.length][
102  Arrays.stream( l_rows )
103  .map( i -> Arrays.stream( i.trim().split( ",|\\s" ) )
104  .map( String::trim )
105  .filter( j -> !j.isEmpty() )
106  .mapToDouble( Double::parseDouble )
107  .boxed()
108  .collect( Collectors.toList() )
109  )
110  .mapToInt( i -> {
111  l_matrix.add( i );
112  return i.size();
113  } )
114  .max()
115  .orElseThrow( () -> new CRuntimeException( p_context ) )
116  ];
117 
118  IntStream.range( 0, l_return.length )
119  .boxed()
120  .forEach( i -> IntStream.range( 0, l_return[i].length )
121  .boxed()
122  .forEach( j -> l_return[i][j] = l_matrix.get( i ).get( j ) ) );
123 
124  return l_return;
125  }
