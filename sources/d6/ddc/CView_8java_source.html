291  @SafeVarargs
292  private final Stream<IView<T>> walkdown( final IPath p_path, final IViewGenerator<T>... p_generator )
293  {
294  if ( ( p_path == null ) || ( p_path.isEmpty() ) )
295  return Stream.of( this );
296 
297  final IView<T> l_view;
298  final String l_root = p_path.get( 0 );
299  synchronized ( this )
300  {
301  // add is run here for avoid overwriting view with a new object reference
302  l_view = m_beliefbase.add(
303  m_beliefbase.viewOrDefault(
304  l_root,
305 
306  p_generator == null || p_generator.length == 0
307  ? null
308  : p_generator[0].apply( l_root, this )
309  )
310  );
311  }
312  if ( l_view == null )
313  throw new CIllegalArgumentException( CCommon.languagestring( this, "notfound", l_root, this.path() ) );
314 
315  return Stream.concat(
316  Stream.of( this ),
317  l_view.walk( p_path.getSubPath( 1 ), p_generator )
318  );
319  }
