320  @Override
321  public final IFuzzyValue<Boolean> trigger( final ITrigger p_trigger, final boolean... p_immediately )
322  {
323  if ( m_sleepingcycles.get() > 0 )
324  return CFuzzyValue.from( false );
325 
326  // check if literal does not store any variables
327  if ( p_trigger.getLiteral().hasVariable() )
328  throw new CIllegalArgumentException( org.lightjason.agentspeak.common.CCommon.languagestring( this, "literalvariable", p_trigger ) );
329 
330  // run plan immediatly and return
331  if ( ( p_immediately != null ) && ( p_immediately.length > 0 ) && ( p_immediately[0] ) )
332  return this.execute( this.generateexecution( Stream.of( p_trigger ) ) );
333 
334  // add trigger for the next cycle must be synchronized to avoid indeterministic state during execution
335  synchronized ( this )
336  {
337  m_trigger.putIfAbsent( p_trigger.contenthash(), p_trigger );
338  }
339  return CFuzzyValue.from( true );

344  public T call() throws Exception
345  {
346  // run beliefbase update, because environment can be changed and decrement sleeping value
347  m_beliefbase.update( (T) this );

361  m_cycletime.set( System.nanoTime() );
362 
363  return (T) this;

394  private Collection<Pair<Triple<IPlan, AtomicLong, AtomicLong>, IContext>> generateexecution( final Stream<ITrigger> p_trigger )
395  {
396  return p_trigger
397  .filter( Objects::nonNull )
398  .flatMap( i -> {
399  final Collection<Triple<IPlan, AtomicLong, AtomicLong>> l_plans = m_plans.get( i );
400 
401  return l_plans == null
402  ? Stream.of()
403  : l_plans
404  .parallelStream()
405 
406  // tries to unify trigger literal and filter of valid unification (returns set of unified variables)
407  .map( j -> new ImmutablePair<>( j, CCommon.unifytrigger( m_unifier, i, j.getLeft().getTrigger() ) ) )
408  .filter( j -> j.getRight().getLeft() )
409 
410  // initialize context
411  .map( j -> new ImmutablePair<>(
412  j.getLeft(),
413  CCommon.instantiateplan(
414  j.getLeft().getLeft(),
415  j.getLeft().getMiddle().get(),
416  j.getLeft().getRight().get(),
417  this,
418  j.getRight().getRight()
419  ).getRight()
420  ) )
421 
422  // check plan condition
423  .filter( j -> m_fuzzy.getDefuzzyfication().defuzzify( j.getLeft().getLeft().condition( j.getRight() ) ) );
424  }
425  )
426  // collectors-call must be toList not toSet because plan-execution can be have equal elements
427  // so a set avoid multiple plan-execution
428  .collect( Collectors.toList() );
429 
430  }

438  private IFuzzyValue<Boolean> execute( final Collection<Pair<Triple<IPlan, AtomicLong, AtomicLong>, IContext>> p_execution )
439  {
440  // update executable plan list, so that test-goals are defined all the time
441  p_execution.parallelStream().forEach( i -> m_runningplans.put(
442  i.getLeft().getLeft().getTrigger().getLiteral().fqnfunctor(),
443  i.getLeft().getLeft().getTrigger().getLiteral().unify( i.getRight() )
444  ) );
445 
446  // execute plan and return values and return execution result
447  return p_execution.parallelStream().map( i -> {
448 
449  final IFuzzyValue<Boolean> l_result = i.getLeft().getLeft().execute( i.getRight(), false, null, null, null );
450  if ( m_fuzzy.getDefuzzyfication().defuzzify( l_result ) )
451  // increment successful runs
452  i.getLeft().getMiddle().getAndIncrement();
453  else
454  // increment failed runs and create delete goal-event
455  i.getLeft().getRight().getAndIncrement();
456 
457  return l_result;
458  } ).collect( m_fuzzy.getResultOperator() );
459  }

467  private boolean active( final boolean p_immediatly )
468  {
469  // if the sleeping time ends or the agent will wakedup by a hard call,
470  // create the trigger and reset the time value
471  if ( ( m_sleepingcycles.compareAndSet( 0, Long.MIN_VALUE ) ) || p_immediatly )
472  {
473  (
474  m_sleepingterm.isEmpty()
475 
476  ? Stream.of( CTrigger.from(
477  ITrigger.EType.ADDGOAL, CLiteral.from(
478  "wakeup"
479  )
480  ) )
481 
482  : m_sleepingterm.parallelStream()
483  .map( i -> CTrigger.from(
484  ITrigger.EType.ADDGOAL,
485  CLiteral.from( "wakeup", i )
486  ) )
487 
488  ).forEach( i -> m_trigger.put( i.contenthash(), i ) );
489 
490  m_sleepingterm.clear();
491  }
492 
493  // if the sleeping time is not infinity decrese the counter
494  if ( ( m_sleepingcycles.get() > 0 ) && ( m_sleepingcycles.get() != Long.MAX_VALUE ) )
495  m_sleepingcycles.decrementAndGet();
496 
497  return m_sleepingcycles.get() <= 0;
498  }
