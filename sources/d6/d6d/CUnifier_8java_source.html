96  public final IFuzzyValue<Boolean> parallel( final IContext p_context, final ILiteral p_literal, final long p_variablenumber,
97  final IExpression p_expression
98  )
99  {

101  final List<Set<IVariable<?>>> l_variables = this.unify( p_context.agent(), p_literal, p_variablenumber );
102  if ( l_variables.isEmpty() )
103  return CFuzzyValue.from( false );
104 
105  // otherwise the expression must be checked, first match will be used
106  final Set<IVariable<?>> l_result = l_variables.parallelStream()
107  .filter( i -> {
108  final List<ITerm> l_return = new LinkedList<>();

169  @Override
170  public final Set<IVariable<?>> literal( final ILiteral p_literal, final ILiteral p_value )
171  {
172  final Set<IVariable<?>> l_result = new HashSet<>();
173 
174  // try to unify exact or if not possible by recursive on the value set
175  if ( !(
176  p_literal.valuehash() == p_value.valuehash()
177  ? m_hashbased.unify(
178  l_result,
179  CCommon.recursiveterm( p_value.orderedvalues() ),
180  CCommon.recursiveterm( p_literal.orderedvalues() )
181  )
182  : m_recursive.unify(
183  l_result,
184  p_value.orderedvalues(),
185  p_literal.orderedvalues()
186  ) ) )
187  return Collections.<IVariable<?>>emptySet();
188 
189  // try to unify exact or if not possible by recursive on the annotation set
190  if ( !(
191  p_literal.annotationhash() == p_value.annotationhash()
192  ? m_hashbased.unify(
193  l_result,
194  CCommon.recursiveliteral( p_value.annotations() ),
195  CCommon.recursiveliteral( p_literal.annotations() )
196  )
197  : m_recursive.unify(
198  l_result,
199  p_value.annotations(),
200  p_literal.annotations()
201  ) ) )
202  return Collections.<IVariable<?>>emptySet();
203 
204  return l_result;
205  }

240  @SuppressWarnings( "unchecked" )
241  private List<Set<IVariable<?>>> unify( final IAgent<?> p_agent, final ILiteral p_literal, final long p_variablenumber )
242  {
243  return p_agent.beliefbase()
244  .stream( p_literal.negated(), p_literal.fqnfunctor() )
245  .parallel()
246  .filter( i -> ( i.emptyValues() == p_literal.emptyValues() ) && ( i.emptyAnnotations() == p_literal.emptyAnnotations() ) )
247  .map( i -> this.literal( (ILiteral) p_literal.deepcopy(), i ) )
248  .filter( i -> p_variablenumber == i.size() )
249  .collect( Collectors.toList() );
250  }
