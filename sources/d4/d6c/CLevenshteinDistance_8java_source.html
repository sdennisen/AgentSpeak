94  private double levenshtein( final String p_first, final String p_second )
95  {
96  // the array of distances
97  double[] l_cost = IntStream.range( 0, p_first.length() + 1 ).mapToDouble( i -> i ).toArray();
98  double[] l_newcost = new double[l_cost.length];
99 
100  for ( int j = 1; j < p_second.length() + 1; j++ )
101  {
102  l_newcost[0] = j;
103 
104  // calculate cost of operation for all characters
105  for ( int i = 1; i < l_cost.length; i++ )
106  l_newcost[i] = min(
107  l_cost[i - 1] + ( p_first.charAt( i - 1 ) == p_second.charAt( j - 1 ) ? 0 : m_replaceweight ),
108  l_newcost[i - 1] + m_deleteweight,
109  l_cost[i] + m_insertweight
110  );
111 
112  final double[] l_swap = l_cost;
113  l_cost = l_newcost;
114  l_newcost = l_swap;
115  }
116 
117  return l_cost[p_first.length()];
118  }
