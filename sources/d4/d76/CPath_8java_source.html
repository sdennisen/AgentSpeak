402  private synchronized void normalize()
403  {
404  if ( m_path.isEmpty() )
405  return;
406 
407  // create path-copy and nomalize (remove dot, double-dot and empty values)
408  final List<String> l_dotremove = m_path.stream()
409  .filter( i -> ( i != null ) && ( !i.isEmpty() ) && ( !".".equals( i ) ) )
410  .collect( Collectors.toList() );
411  if ( l_dotremove.isEmpty() )
412  return;
413 
414  final String l_last = l_dotremove.get( l_dotremove.size() - 1 );
415  final List<String> l_backremove = IntStream.range( 0, l_dotremove.size() - 1 )
416  .boxed()
417  .filter( i -> !l_dotremove.get( i + 1 ).equals( ".." ) )
418  .map( l_dotremove::get )
419  .collect( Collectors.toList() );
420  if ( !"..".equals( l_last ) )
421  l_backremove.add( l_last );
422 
423  // clear internal path and add optimized path
424  m_path.clear();
425  l_backremove.forEach( m_path::add );
426  }
