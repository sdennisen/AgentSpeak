66  final List<ITerm> p_annotation
67  )
68  {
69  // first parameter is a list with elements, which will return by the selection
70  // second parameter is a numeric value for each element
71  final List<?> l_items = p_argument.get( 0 ).raw();
72  final List<Double> l_weight = this.weight(
73  l_items,
74  p_argument.get( 1 ).<List<?>>raw().stream()
75  // list can be contains default Java objects or term objects
76  .map( i -> i instanceof ITerm ? ( (ITerm) i ).<Number>raw() : (Number) i )
77  .map( Number::doubleValue )
78  .map( Math::abs ),
79  p_argument.subList( 2, p_argument.size() ),
80  p_annotation
81  );
82 
83  if ( ( l_items.isEmpty() ) || ( l_items.size() != l_weight.size() ) )
84  return CFuzzyValue.from( false );
85 
86  // select a random value and scale with the sum
87  double l_random = m_random.nextDouble() * l_weight.stream().mapToDouble( i -> i ).sum();
88  for ( int i = 0; i < l_weight.size(); i++ )
89  {
90  l_random -= l_weight.get( i );
91  if ( l_random <= 0 )
92  {
93  p_return.add( CRawTerm.from( l_items.get( i ) ) );
94  return CFuzzyValue.from( true );
95  }
96  }
97 
98  // on rounding error return last element
99  p_return.add( CRawTerm.from( l_items.get( l_items.size() - 1 ) ) );
100  return CFuzzyValue.from( true );
101  }
